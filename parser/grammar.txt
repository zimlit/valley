expression = literal
           | unary
           | binary
           | grouping :

literal = INT | FLOAT | STRING | "true" | "false" ;
grouping = "(" expression ")"
unary = ( "-" | "~" | "!" ) expression
binary = expression operator expression
operator = "==" | "!=" | "<" | "<=" | ">" | ">="
         | "+" | "-" | "*" | "/"
